\section{Application}
\begin{enumerate}[label=\thesection.\arabic*
,ref=\thesection.\theenumi]
%
\item The Steinhartâ€“Hart equation is a model of the resistance of a 
thermistor at different temperatures. The equation is given by
\begin{equation}
\label{eq:ls_steiner}
\frac{1}{ \tau}=w_1+w_2\ln(R)+w_3[\ln(R)]^{3}
\end{equation}
%
Let
\begin{align}
\vec{a}_1 &= \myvec{1 \\ \ln(R_1)\\\sbrak{\ln(R_1)}^{3}}
\\
{y}_1 &=  \frac{1}{\tau_1}
\\
\vec{w} &= \myvec{w_1\\ w_2\\w_3 }
\end{align}
%
Show that
\begin{equation}
{y}_1 =  \vec{a}_1^T\vec{w}
\end{equation}
\item Suppose for $n > 3$
\begin{align}
\vec{y} = \myvec{y_1 \\ y_2\\ \vdots \\ y_n},
A^T  = \myvec{\vec{a}_1& \vec{a}_2 & \vdots &\vec{a}_n },
\end{align}
%
show that 
\begin{equation}
\vec{y} = A\vec{w}
\end{equation}
%
\item For $\tau = 10^{\degree} C - 100^{\degree} C$, use the PT-100 resistance table in
\begin{lstlisting}
https://github.com/gadepall/EE1390/blob/master/refs/5pt100sensoren_e.pdf?raw=true
\end{lstlisting}
to estimate $\vec{w}$ 
using 
the relation
\begin{equation}
\hat{\vec{w}} = \brak{A^TA}^{-1}A^T\vec{y}
\end{equation}
\item Verify your result by finding the temperature when the resistance is $175.86\Omega$.
\end{enumerate}
